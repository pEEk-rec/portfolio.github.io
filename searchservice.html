#include <iostream>
#include <unordered_map>
#include <vector>
using namespace std;

// Service structure
struct Service {
    string name;
    int number;
    double rating;
};

int main() {
    // Data: city services
    unordered_map<string, Service> cityServices = {
        {"hospital", {"Hospitals", 10, 4.5}},
        {"police station", {"Police Stations", 5, 4.8}},
        {"school", {"Schools", 15, 4.3}},
        {"supermarket", {"Supermarkets", 20, 4.4}},
        {"shopping mall", {"Shopping Malls", 7, 4.6}},
        {"restaurant", {"Restaurants", 25, 4.2}},
        {"fire station", {"Fire Stations", 3, 4.7}},
        {"local market", {"Local Markets", 12, 4.1}},
        {"pharmacy", {"Pharmacies", 18, 4.6}},
        {"hardware store", {"Hardware Stores", 6, 4.0}},
        {"grocery store", {"Grocery Stores", 22, 4.3}},
        {"university", {"Universities", 4, 4.9}},
        {"bank", {"Banks", 10, 4.5}},
        {"transportation", {"Transportation Services", 8, 4.2}},
        {"utility service", {"Utility Services", 6, 4.4}},
        {"public park", {"Public Parks", 10, 4.8}},
        {"business service", {"Business Services", 5, 4.0}},
        {"public library", {"Public Libraries", 3, 4.7}},
        {"gas station", {"Gas Stations", 9, 4.3}},
        {"museum", {"Museums", 2, 4.9}},
        {"multiplex", {"Multiplexes", 4, 4.5}},
        {"public toilet", {"Public Toilets", 15, 4.0}},
        {"water service", {"Water Services", 4, 4.6}}
    };

    // User query
    string query;
    cout << "Enter the type of service you are looking for: ";
    getline(cin, query);

    // Convert query to lowercase for case-insensitive search
    for (char& c : query) c = tolower(c);

    // Search for the service
    if (cityServices.find(query) != cityServices.end()) {
        Service foundService = cityServices[query];
        cout << "\nService Found: " << foundService.name << endl;
        cout << "Number of locations: " << foundService.number << endl;
        cout << "Average rating: " << foundService.rating << endl;
    } else {
        cout << "\nService not found. Please check the spelling or try a different query." << endl;
    }

    return 0;
}
